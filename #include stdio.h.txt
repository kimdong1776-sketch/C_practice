#include <stdio.h>
#include <stdlib.h>

// [0]번째 열을 기준으로 정렬
int compare_dynamic(const void *a, const void *b) {
    // a와 b는 '포인터(int*)의 주소'를 가리킴. 따라서 int** 로 형변환!
    int *row1 = *(int **)a;
    int *row2 = *(int **)b;
    
    return row1[0] - row2[0];
}

int main() {
    int rows = 3;
    // 1. 이중 포인터 동적 할당
    int **arr = (int **)malloc(sizeof(int *) * rows);
    for (int i = 0; i < rows; i++) {
        arr[i] = (int *)malloc(sizeof(int) * 2);
    }
    
    arr[0][0] = 30; arr[0][1] = 5;
    arr[1][0] = 10; arr[1][1] = 9;
    arr[2][0] = 20; arr[2][1] = 1;

    // 행의 개수: 3, 데이터 하나의 크기: 포인터의 크기(sizeof(int*))
    qsort(arr, rows, sizeof(int *), compare_dynamic);

    for (int i = 0; i < rows; i++) {
        printf("[%d, %d]\n", arr[i][0], arr[i][1]);
    }

    // 메모리 해제 생략...
    return 0;
}
